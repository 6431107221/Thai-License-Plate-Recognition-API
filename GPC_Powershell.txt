gcloud config set project [PROJECT_ID]

gcloud auth configure-docker

docker build -t gcr.io/[PROJECT-ID]/thai-lpr-api .

docker push gcr.io/[PROJECT-ID]/thai-lpr-api

gcloud run deploy thai-lpr-api `
  --image gcr.io/[PROJECT-ID]/thai-lpr-api `
  --region asia-southeast1 `
  --platform managed `
  --allow-unauthenticated `
  --memory 2Gi `
  --cpu 1 `
  --max-instances 1 `
  --min-instances 0 `
  --no-cpu-throttling
